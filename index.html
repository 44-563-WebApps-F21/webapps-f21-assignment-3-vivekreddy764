Algorithm

The Prüfer code is a way of encoding a labeled tree with n vertices using a sequence of n−2 integers in the interval [0;n−1]. This encoding also acts as a bijection between all spanning trees of a complete graph and the numerical sequences. Although using the Prüfer code for storing and operating on tree is impractical due the specification of the representation, the Prüfer codes are used frequently: mostly in solving combinatorial problems. The inventor - Heinz Prüfer - proposed this code in 1918 as a proof for Cayley's formula. 
        
            vector(vector(int)) adj;

                vector(int) pruefer_code() {
                    int n = adj.size();
                    set(int) leafs;
                    vector(int) degree(n);
                    vector(bool) killed(n, false);
                    for (int i = 0; i < n; i++) {
                        degree[i] = adj[i].size();
                        if (degree[i] == 1)
                            leafs.insert(i);
                    }
                
                    vector(int) code(n - 2);
                    for (int i = 0; i < n - 2; i++) {
                        int leaf = *leafs.begin();
                        leafs.erase(leafs.begin());
                        killed[leaf] = true;
                
                        int v;
                        for (int u : adj[leaf]) {
                            if (!killed[u])
                                v = u;
                        }
                
                        code[i] = v;
                        if (--degree[v] == 1)
                            leafs.insert(v);
                    }
                
                    return code;
                }
